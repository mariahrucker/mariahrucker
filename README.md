import java.util.HashMap;
import java.util.Map;
import java.util.Scanner;

public class StarWarsScrabbleGame {
    public static void main(String[] args) {
        // Initialize the letter scores
        Map<Character, Integer> letterScores = new HashMap<>();
        letterScores.put('a', 1);
        letterScores.put('b', 3);
        letterScores.put('c', 3);
        letterScores.put('d', 2);
        letterScores.put('e', 1);
        letterScores.put('f', 4);
        letterScores.put('g', 2);
        letterScores.put('h', 4);
        letterScores.put('i', 1);
        letterScores.put('j', 8);
        letterScores.put('k', 5);
        letterScores.put('l', 1);
        letterScores.put('m', 3);
        letterScores.put('n', 1);
        letterScores.put('o', 1);
        letterScores.put('p', 3);
        letterScores.put('q', 10);
        letterScores.put('r', 1);
        letterScores.put('s', 1);
        letterScores.put('t', 1);
        letterScores.put('u', 1);
        letterScores.put('v', 4);
        letterScores.put('w', 4);
        letterScores.put('x', 8);
        letterScores.put('y', 4);
        letterScores.put('z', 10);

        // Prompt the player for a word
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a word: ");
        String word = scanner.nextLine().toLowerCase();

        // Calculate the score of the word
        int score = 0;
        for (char letter : word.toCharArray()) {
            if (letterScores.containsKey(letter)) {
                score += letterScores.get(letter);
            }
        }

        // Display the score to the player
        System.out.println("Your score for \"" + word + "\" is: " + score);
    }
}
